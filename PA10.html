def calc_I(Is,Ib,V,Gp,Vb):
    p1 = Is*(np.exp((1.2*V)/0.025))
    p2 = Gp*V
    p3 = Ib*np.exp((-1.2/0.025)*(V+Vb))
    return p1 + p2 - p3

def nonl_fit(x,a,b,c,d):
    pA = a*(np.exp((1.2*x/25e-3)) -1 )
    pB = b*x
    pC = c*(np.exp(1.2*(-(x+d)/25e-3)) -1)
    
    return pA + pB - pC
